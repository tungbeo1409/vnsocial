rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - Rules cho tính năng kết bạn
    match /users/{userId} {
      // Cho phép đọc tất cả user profiles (để tìm kiếm)
      allow read: if true;
      
      // Cho phép user tự tạo profile của mình
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Cho phép update:
      // 1. User tự update profile của mình (bất kỳ field nào)
      // 2. User update friendRequests, friends, sentRequests (khi gửi/nhận/hủy lời mời)
      allow update: if request.auth != null && (
        // Trường hợp 1: User tự update profile của mình
        request.auth.uid == userId ||
        // Trường hợp 2: User update các field liên quan đến friends
        // Cho phép nếu chỉ update friendRequests, friends, sentRequests
        // VÀ user đang update chính profile của mình HOẶC đang update profile người khác nhưng chỉ thêm/xóa từ arrays
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['friendRequests', 'friends', 'sentRequests']) &&
         // Cho phép nếu user đang update profile của mình
         (request.auth.uid == userId ||
          // HOẶC nếu đang update friendRequests của người khác (khi nhận lời mời)
          request.resource.data.friendRequests.hasAny([request.auth.uid]))
        )
      );
      
      // Cho phép user tự xóa profile
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Posts collection
    match /posts/{postId} {
      // Cho phép đọc tất cả posts
      allow read: if true;
      // Chỉ user đã đăng nhập mới được tạo post
      allow create: if request.auth != null;
      // Chỉ user đã đăng nhập mới được update (like, comment)
      allow update: if request.auth != null;
      // Chỉ chủ post mới được xóa
      allow delete: if request.auth != null && 
        resource.data.userId == request.auth.uid;
    }
  }
}
